type tp =
  1. | TVoid
    (*

    void

    *)
  2. | TFloat
    (*

    float

    *)
  3. | TInteger of int
    (*

    i1, i2 ... in

    *)
  4. | TPointer
    (*

    ptr

    *)
  5. | TVector of int * tp
    (*

    <int x primitive_type>

    *)
  6. | TArr of int * tp
    (*

    int x type

    *)
  7. | TStruct of tp list
    (*

    {type1, type2...}

    *)
  8. | TLabel
    (*

    label

    *)
  9. | TFunc of tp * tp list
    (*

    <returntype> (<parameter list>)

    *)

type const =
  1. | CVoid
  2. | CInteger of int * Stdlib.Int64.t
    (*

    size and value

    *)
  3. | CFloat of float
  4. | CPointer of pointer_const
  5. | CVector of const list
    (*

    <const, const, ...>

    *)
  6. | CArr of const list
    (*

    const, const, ...

    *)
  7. | CStruct of const list
    (*

    {const, const, ...}

    *)
  8. | CLabel of basic_block
  9. | CFunc of func
and value =
  1. | FromVariable of variable * tp
  2. | Const of const
and terminator_instruction =
  1. | Ret of tp * value
    (*

    ret <type> <value>

    *)
  2. | Br of value
    (*

    br label <dest>

    *)
  3. | BrCond of value * value * value
    (*

    br i1 <cond>, label <iftrue>, label <iffalse>

    *)
  4. | Switch of tp * value * value * (value * value) list
    (*

    switch <intty> <value>, label <defaultdest> <intty> <val>, label <dest> ...

    *)
  5. | Unreachable
    (*

    unreachable

    *)
and unary_operation =
  1. | Fneg of variable * tp * value
    (*

    <result> = fneg <ty> <op1>

    *)
and binary_operation_body = variable * tp * value * value
and binary_operation =
  1. | Add of binary_operation_body
  2. | Fadd of binary_operation_body
  3. | Mul of binary_operation_body
  4. | Fmul of binary_operation_body
  5. | Sub of binary_operation_body
  6. | Fsub of binary_operation_body
  7. | Udiv of binary_operation_body
  8. | Sdiv of binary_operation_body
  9. | Fdiv of binary_operation_body
  10. | Urem of binary_operation_body
  11. | Srem of binary_operation_body
  12. | Frem of binary_operation_body
and bitwise_binary_operation =
  1. | Shl of binary_operation_body
  2. | Lshr of binary_operation_body
  3. | Ashr of binary_operation_body
  4. | And of binary_operation_body
  5. | Or of binary_operation_body
  6. | Xor of binary_operation_body
and vector_instruction =
  1. | Extractelement of variable * tp * value * tp * value
    (*

    <result> = extractelement <n x <ty>> <val>, <ty2> <idx>

    *)
  2. | Insertelement of variable * tp * value * value * tp * value
    (*

    <result> = insertelement <n x <ty>> <val>, <ty> <elt>, <ty2> <idx>

    *)
  3. | Shufflevector of variable * tp * value * value * int * const
    (*

    <result> = shufflevector <n x <ty>> <v1>, <n x <ty>> <v2>, <m x i32> <mask>

    *)
and aggregate_instruction =
  1. | Extractvalue of variable * tp * value * int list
    (*

    <result> = extractvalue <aggregate type> <val>, <idx>{, <idx>}*

    *)
  2. | Insertvalue of variable * tp * value * tp * value * int list
    (*

    <result> = insertvalue <aggregate type> <val>, <ty> <elt>, <idx>{, <idx>}*

    *)
and memory_address_instruction =
  1. | Alloca of variable * tp * value * align
    (*

    <result> = alloca <type> , <ty> <NumElements> , align <alignment>

    *)
  2. | Store of tp * value * value * align
    (*

    store <ty> <value>, ptr <pointer>, align <alignment>

    *)
  3. | Load of variable * tp * value * align
    (*

    <result> = load <ty>, ptr <pointer>, align <alignment>

    *)
  4. | Getelementptr of variable * tp * tp * value * (tp * value) list
    (*

    <result> = getelementptr <ty>, <ptr_or_ptr_vector> <ptrval>{, <ty> <idx>}*

    *)
and conversion_instruction_body = variable * tp * value * tp

<result> = <conv_inst> <ty> <value> to <ty2>

and conversion_instruction =
  1. | TruncTo of conversion_instruction_body
  2. | ZextTo of conversion_instruction_body
  3. | SextTo of conversion_instruction_body
  4. | FptouiTo of conversion_instruction_body
  5. | FptosiTo of conversion_instruction_body
  6. | UitofpTo of conversion_instruction_body
  7. | SitofpTo of conversion_instruction_body
  8. | PrttointTo of conversion_instruction_body
  9. | InttoprtTo of conversion_instruction_body
and other_operation =
  1. | Icmp of variable * string * tp * value * value
    (*

    <result> = icmp <cond> <ty> <op1>, <op2>

    *)
  2. | Fcmp of variable * string * tp * value * value
    (*

    <result> = fcmp <cond> <ty> <op1>, <op2>

    *)
  3. | Phi of variable * tp * (value * value) list
    (*

    <result> = phi <ty> <val0>, <label0>, ...

    *)
  4. | Select of variable * tp * value * tp * value * value
    (*

    <result> = select <i1 or i1_vec> <cond>, <ty> <val1>, <ty> <val2>

    *)
  5. | Call of variable * tp * value * value list
    (*

    <result> = call <ty> <fnptrval>(<function args>)

    *)
and instruction =
  1. | Terminator of terminator_instruction
  2. | Unary of unary_operation
  3. | Binary of binary_operation
  4. | BitwiseBinary of bitwise_binary_operation
  5. | Vector of vector_instruction
  6. | Aggregate of aggregate_instruction
  7. | Other of other_operation
  8. | MemoryAddress of memory_address_instruction
  9. | Conversion of conversion_instruction
and basic_block = instruction list
and func = {
  1. ftp : tp;
  2. parameters : variable list;
  3. basic_blocks : (variable * const) list;
}
